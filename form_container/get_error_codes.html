Method Form_Container->get_error_codes

Parameters: none

Returns: array

Returns all error codes for the messages added for the container.

$form->example->add_error('Lorem', 'foo');
$form->example->add_error('Ipsum', 'foo');
$form->example->add_error('Dolor', 'bar');
$form->example->add_error('Sit Amet', 'bar');

print_r($form->example->get_error_codes());
// Array
// (
//     [0] => foo
//     [1] => bar
// )
