<?php

require('./lib/form.php');



$form = Form::from_request('foo[bar][baz]');


$form->set_defaults(array(
    'subs' => array(),
    'multi' => array(),
    'remember' => 0,
));



function inspect(Form_Container $form) {

    if (!count($form)) {
        return;
    }
    print('<ul>');
    foreach ($form as $name => $input) {
        print('<li><b>'.htmlspecialchars($name).'</b> ');
        print('<small>('.htmlspecialchars($input->get_name()).')</small>: ');


        if (count($input)) {
            // complex input
            inspect($input);
        } else {
            // single value
            print($input->as_string());
        }


    }
    print('</ul>');
}
